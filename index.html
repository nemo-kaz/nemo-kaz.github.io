Consoleオブジェクト

開発者ツール(F12)のコンソールで表示を確認してください。

console.dirxml()はブラウザ(バージョン)によって表示が異なることがあります。

  1. dirxml用サンプル
  2. コンソールにDOM要素として表示されます。
      console.group('ログ出力');
      console.debug('debug メッセージ');
      console.error('error メッセージ');
      console.info('info メッセージ');
      console.log('log メッセージ');
      console.warn('warn メッセージ');
      console.groupEnd();

      console.group('特殊な出力');
      /* dirxmlによるHTML要素の表示 */
      console.dirxml(document.getElementById('ol'));
      /* traceによるスタックトレースの表示 */
      function trace_func1() {
        trace_func2();
      }
      function trace_func2() {
        console.trace();
      }
      trace_func1();
      /* assert */
      console.assert(false,'assert false メッセージ');
      console.assert(true ,'assert true メッセージ');
      console.groupEnd();
